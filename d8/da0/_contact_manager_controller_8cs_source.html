64 [HttpPost("CreateContacts"), Authorize(AuthenticationSchemes = JwtBearerDefaults.AuthenticationScheme)]
65 public async Task<ControllerResult<IEnumerable<ContactVM>>> CreateContact([FromBody] List<ContactVM> contactVMList)
66 {
67 var validationResult = contactVMList.FirstOrDefault().Validate(_sessionService.SessionGetClientName());
68 if (validationResult.Success)
69 {
70 if (_sessionService.SessionGetClientName() == "RJ")
71 {
72 // check if Address1 is empty.
73 var contact = contactVMList.FirstOrDefault();
74 var errors = new List<KeyValuePair<string, string>>();
75 if (string.IsNullOrEmpty(contact.Address1))
76 {
77 errors.Add(new KeyValuePair<string, string>("Address1", "Required"));
78 }
79
80 if (string.IsNullOrEmpty(contact.Title))
81 {
82 errors.Add(new KeyValuePair<string, string>("Title", "Required"));
83 }
84
85 if (errors.Count > 0)
86 {
87 return new ApiResponse<IEnumerable<ContactVM>>(errors).AsResult(ErrorType.InputValidationFailure);
88 }
89 }
90
91 return await service.CreateContactMF(contactVMList);
92 }
93 else
94 {
95 if (_sessionService.SessionGetClientName() == "RJ")
96 {
97 /* need to remove
98 "{Key: Address1, Value: Check property does not contain any value}",
99 "{Key: Email, Value: Check property does not contain any value}"
100 and add
101 "{Key: Address1, Value: Required}",
102 */
103 var addAddress1Error = false;
104 var errorsToRemove = new List<KeyValuePair<string, string>>();
105
106 foreach (var item in validationResult.Errors)
107 {
108 if (item.Key == "Address1" && item.Value == "Check property does not contain any value")
109 {
110 addAddress1Error = true;
111 errorsToRemove.Add(item);
112 }
113 else if (item.Key == "Email" && item.Value == "Check property does not contain any value")
114 {
115 errorsToRemove.Add(item);
116 }
117 else if (item.Key == "LastName" && item.Value == "Required")
118 {
119 errorsToRemove.Add(item);
120 }
121 }
122
123 if (errorsToRemove.Count > 0)
124 {
125 foreach (var item in errorsToRemove)
126 {
127 _ = validationResult.Errors.Remove(item);
128 }
129 }
130
131 if (addAddress1Error)
132 {
133 validationResult.Errors.Add(new KeyValuePair<string, string>("Address1", "Required"));
134 }
135
136 if (validationResult.Errors.Count == 0)
137 {
138 return await service.CreateContactMF(contactVMList);
139 }
140 }
141
142 return new ApiResponse<IEnumerable<ContactVM>>(validationResult.Errors).AsResult(ErrorType.InputValidationFailure);
143 }
