65 private string SetCustomFieldValue(string customFieldType, string customFieldValue)
66 {
67 if (customFieldType.Equals("date", StringComparison.CurrentCultureIgnoreCase))
68 {
69 return EncodeInputString(DateTime.Parse(customFieldValue).ToString("yyyy-MM-dd"));
70 }
71 else if (customFieldType.Equals("datetime", StringComparison.CurrentCultureIgnoreCase))
72 {
73 return EncodeInputString(DateTime.Parse(customFieldValue).ToString("yyyy-MM-ddTHH:mm:ssZ"));
74 }
75 else if (NonQuotedDataTypes.Contains(customFieldType, StringComparer.InvariantCultureIgnoreCase))
76 {
77 return EncodeInputString(customFieldValue);
78 }
79 else
80 {
81 return $"'{EncodeInputString(customFieldValue)}'";

85 public async Task<ServiceResult<List<SFQueryContactVM>>> GetGetSalesForceContacts(SalesForceGetContactInput salesForceGetContactInput, string instanceUrl, string accessToken, string apiVersion)
86 {
87 var client = new ForceClient(instanceUrl, accessToken, apiVersion);
88
89 var whereExpressions = new List<string>();
90
91 if (!string.IsNullOrWhiteSpace(salesForceGetContactInput.State))
92 {
93 _ = EncodeInputString(salesForceGetContactInput.State);
94
95 whereExpressions.Add($"MailingState = '{EncodeInputString(salesForceGetContactInput.State)}'");
96
97 }
98
99 if (!string.IsNullOrEmpty(salesForceGetContactInput.AccountName))
100 {
101 whereExpressions.Add($"Name LIKE '%{salesForceGetContactInput.AccountName}%'");
102 }
103
104 if (!string.IsNullOrEmpty(salesForceGetContactInput.LeadSource))
105 {
106 whereExpressions.Add($"LeadSource = '{salesForceGetContactInput.LeadSource}'");
107 }
108
109 if (salesForceGetContactInput.ExcludeNoEmail)
110 {
111 whereExpressions.Add("Email != ''");
112 }
113
114 if (!string.IsNullOrEmpty(salesForceGetContactInput.CustomFieldId) && !string.IsNullOrEmpty(salesForceGetContactInput.CustomFieldValue))
115 {
116 var getCustomFieldWhereExpression = await GetCustomFieldWhereExpression(salesForceGetContactInput.CustomFieldId, salesForceGetContactInput.CustomFieldValue, instanceUrl, accessToken);
117 if (!string.IsNullOrWhiteSpace(getCustomFieldWhereExpression.ErrorMessage))
118 {
119 return new ServiceResult<List<SFQueryContactVM>>(null, getCustomFieldWhereExpression.ErrorMessage);
120 }
121
122 whereExpressions.Add(getCustomFieldWhereExpression.Data);
123 }
124
125 var sqlQuery = "SELECT FirstName, LastName, Email, Name, MobilePhone, MailingAddress, Birthdate FROM Contact";
126 if (whereExpressions.Count != 0)
127 {
128 sqlQuery += " WHERE " + string.Join(" AND ", whereExpressions);
129 }
130
131 var result = await client.QueryAsync<SFQueryContactVM>(sqlQuery);
132
133 var contacts = result.Records;
134 return new ServiceResult<List<SFQueryContactVM>>(contacts, "");
135 }

187 private async Task<ServiceResult<string>> GetCustomFieldWhereExpression(string customFieldId, string customFieldValue, string instanceUrl, string accessToken)
188 {
189 var rightOperand = "";
190 if (string.IsNullOrWhiteSpace(customFieldId) || string.IsNullOrWhiteSpace(customFieldValue))
191 {
192 return new ServiceResult<string>("", "");
193 };
194
195 var customFieldsResult = await GetCustomFieldsAdmin(["Contact", "Lead"], instanceUrl, accessToken, ApiVersion);
196 if (!string.IsNullOrWhiteSpace(customFieldsResult.ErrorMessage))
197 {
198 return new ServiceResult<string>(null, customFieldsResult.ErrorMessage);
199 }
200
201 var customField = customFieldsResult.Data.FirstOrDefault(x => x.Id == customFieldId);
202 if (customField is null || customField.Metadata is null || UnsupportedDataTypes.Contains(customField.Metadata.Type, StringComparer.InvariantCultureIgnoreCase))
203 {
204 return new ServiceResult<string>("", "");
205 }
206
207 rightOperand = SetCustomFieldValue(customField.Metadata.Type, customFieldValue);
208
209 return new ServiceResult<string>($"{customField.FullName} = {rightOperand}", "");
210 }
