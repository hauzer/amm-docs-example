111 public async Task<ServiceResult<string>> PublishTweet(SocialContent socialContent, SocialAccount socialAccount, string socialToken)
112 {
113 var twitterAccount = await _dbContext.TwitterAccounts.FirstOrDefaultAsync(ta => ta.Enabled && ta.SocialToken == socialToken && ta.Id == socialAccount.AccountId);
114 if (twitterAccount == null)
115 {
116 twitterAccount = await _dbContext.TwitterAccounts.FirstOrDefaultAsync(ta => ta.Enabled && ta.SocialToken == socialToken);
117 if (twitterAccount == null)
118 {
119 return new ServiceResult<string>(null, "Error: Twitter Account not found.");
120 }
121 }
122
123 var checkTwitterCategory = CheckTwitterCategory(socialContent);
124 switch (checkTwitterCategory)
125 {
126 case "url":
127 var publishUrl = PublishUrl(socialContent?.Text, socialContent?.Link?.Url, twitterAccount);
128 return new ServiceResult<string>(publishUrl.Data, publishUrl.ErrorMessage);
129 case "image":
130 var publishImage = await PublishImage(socialContent?.Picture?.Url, twitterAccount, socialContent?.Text);
131 return new ServiceResult<string>(publishImage.Data, publishImage.ErrorMessage);
132 case "video":
133 var publishVideo = await PublishVideo(socialContent?.Video?.Url, twitterAccount, socialContent?.Text);
134 return new ServiceResult<string>(publishVideo.Data, publishVideo.ErrorMessage);
135 case "text":
136 var publishMessage = PublishMessage(socialContent?.Text, twitterAccount);
137 return new ServiceResult<string>(publishMessage.Data, publishMessage.ErrorMessage);
138 }
139
140 return new ServiceResult<string>(null, "");
141 }

159 return "text";

365 if (user == null)

377 item.AccessTokenSecret = userCreds.AccessTokenSecret;
378 }
379
380 _dbContext.TwitterAccounts.UpdateRange(existingAccounts);
381 _ = await _dbContext.SaveChangesAsync();
382 }
383
384 var accountInfo = new SocialAccountInfoVM()
385 {
386 Provider = SocialProvider.Twitter,
387 FirstName = user.Name,
388 LastName = user.ScreenName,
389 AccessToken = userCreds.AccessToken,
