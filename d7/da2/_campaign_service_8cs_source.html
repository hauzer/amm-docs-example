62 public async Task<ServiceResult<List<CampaignVM>>> GetCampaignsForProfile(CampaignFilterDTO campaignFilterDTO)
63 {
64 var clientId = _sessionService.SessionGetClientId();
65 var user = _sessionService.SessionGetUser();
66 var settingsIsNewDaysExpire = SettingsService.SingleByName("Days_before_Campaign_isNew_expires", clientId);
67 var settingsIsNewDateCriteria = SettingsService.SingleByName("Date_criteria_Campaign_isNew", clientId);
68 //if the campaign filter has any inclusion tags, it is visible only to users that has same tags.
69 var campaigns = await GetList(null, c => c.Name, false, false, null, false, x => x.CampaignSettings);
70
71 if (!string.IsNullOrEmpty(campaigns.ErrorMessage) || campaigns.Data == null)
72 {
73 logger.LogInformation("{ErrorMessage}", campaigns.ErrorMessage);
74 }
75
76 var allCampaigns = campaigns.Data.Where(c => {
77 if (string.IsNullOrWhiteSpace(c.Filter))
78 {
79 return true;
80 }
81
82 CampaignFilter filter;
83 try
84 {
85 filter = JsonHelper.Deserialize<CampaignFilter>(c.Filter);
86 }
87 catch
88 {
89 return false;
90 }
91
92 return
93 filter.Status == "Online" &&
94 (
95 filter.InclusionTags.Count == 0 ||
96 filter.InclusionTags.Any(x => campaignFilterDTO.InclusionTags.Any(y => y == x))
97 );
98 })
99 .ToList();
100
101 foreach (var campaign in allCampaigns.Where(a => !a.IsNew))
102 {
103 switch (settingsIsNewDateCriteria.Data.Value)
104 {
105 case "Created":
106 if (DateTime.Now.Date <= campaign.DateCreated.AddDays(Convert.ToInt32(settingsIsNewDaysExpire.Data.Value)).Date)
107 {
108 campaign.IsNew = true;
109 }
110
111 break;
112
113 case "Modified":
114 if (DateTime.Now.Date <= campaign.DateModified.AddDays(Convert.ToInt32(settingsIsNewDaysExpire.Data.Value)).Date)
115 {
116 campaign.IsNew = true;
117 }
118
119 break;
120
121 default:
122 campaign.IsNew = false;
123 break;
124 }
125 }
126
127 return new ServiceResult<List<CampaignVM>>(allCampaigns, campaigns.ErrorMessage);
128 }

145 //add or edit date in title
146 private static void ParseFeedTitleDate(RSSItemVM rssItemVM, out DateTime dateTime)
147 {
148 var titleArr = rssItemVM.Title.Split("|").ToList();
149 if (DateTime.TryParse(titleArr.Last(), out dateTime))
150 {
151 titleArr[^1] = dateTime.ToString("MM/dd/yy");
152 }
153 else
154 {
155 if (DateTime.TryParse(rssItemVM.PubDate, out dateTime) ||
156 DateTime.TryParseExact(rssItemVM.PubDate, "dd.MM.yyyy", CultureInfo.CurrentCulture, DateTimeStyles.None, out dateTime) ||
157 DateTime.TryParseExact(rssItemVM.PubDate, "MM/dd/yyyy", CultureInfo.CurrentCulture, DateTimeStyles.None, out dateTime) ||
158 DateTime.TryParseExact(rssItemVM.PubDate, "MM/dd/yy", CultureInfo.CurrentCulture, DateTimeStyles.None, out dateTime))
159 {
160 titleArr.Add(dateTime.ToString("MM/dd/yy"));
161 }
162 }
163
164 rssItemVM.Title = string.Join(" | ", titleArr);
165 }

167 //get rss feeds
168 public async Task<ServiceResult<List<RSSItemVM>>> GetRssFeeds(int campaignTemplateId)
169 {
170 var campaignResult = await Single(campaignTemplateId);
171 if (campaignResult.ErrorMessage != "")
172 {
173 return new ServiceResult<List<RSSItemVM>>(null, campaignResult.ErrorMessage);
174 }
175
176 //find query parameter "get"
177 var qscoll = HttpUtility.ParseQueryString(new Uri(campaignResult.Data.Feed).Query);
178 var queryParamGet = -6;
179 if (!string.IsNullOrWhiteSpace(qscoll["get"]))
180 {
181 queryParamGet = int.Parse(qscoll["get"]);
182 }
183
184 var dtToday = DateTime.Today.AddMonths(queryParamGet);
185
186 //check if there is query parameter "filter"
187 var isFiltered = false;
188 if (!string.IsNullOrWhiteSpace(qscoll["filter"]))
189 {
190 isFiltered = true;
191 }
192
193 //download xml with rss items
194 var httpClient = _httpClientFactory.CreateClient("HttpClient");
195 var rssData = await httpClient.GetStringAsync(campaignResult.Data.Feed);
196 var xmlSet = XDocument.Parse(rssData);
197 var xmlItems = xmlSet.Descendants("item");
198 var items = XmlHelper.DeserializeList<RSSItemVM>(xmlItems)?.ToList();
199
200 //exclude all items that do not meet requirements
201 for (var i = items.Count - 1; i >= 0; i--)
202 {
203 ParseFeedTitleDate(items[i], out var dateTime);
204 if (dateTime >= dtToday)
205 {
206 if (isFiltered)
207 {
208 if (items[i].Title.Contains(qscoll["filter"]))
209 {
210 continue;
211 }
212 else
213 {
214 items.RemoveAt(i);
215 }
216 }
217 else
218 {
219 continue;
220 }
221 }
222 else
223 {
224 items.RemoveAt(i);
225 }
226 }
227
228 return new ServiceResult<List<RSSItemVM>>(items, "");
229 }
230 }
