731 private static async Task<IEnumerable<TClass>> getSqlAsync<TClass>(DbConnection conn, string sql)
732 {
733 await conn.OpenAsync();
734 if (conn.State != ConnectionState.Open)
735 {
736 return null;
737 }
738
739 var command = conn.CreateCommand();
740 command.CommandText = sql;
741
742 var reader = await command.ExecuteReaderAsync();
743 var schema = await reader.GetColumnSchemaAsync();
744 var result = new List<TClass>();
745
746 if (schema.Count == 1)
747 {
748 while (await reader.ReadAsync())
749 {
750 // Have to do this type conversion dance, otherwise C# will complain in some cases (see double unboxing).
751 var item = (TClass)Convert.ChangeType(Convert.ChangeType(reader.GetValue(0), reader.GetFieldType(0)), typeof(TClass));
752 if (item == null)
753 {
754 return null;
755 }
756
757 result.Add(item);
758 }
759 }
760 else if (schema.Count > 1)
761 {
762 var type = typeof(TClass);
763 while (await reader.ReadAsync())
764 {
765 var obj = type.GetConstructor(Array.Empty<Type>())?.Invoke(Array.Empty<object>());
766 if (obj == null)
767 {
768 return null;
769 }
770
771 for (var i = 0; i < schema.Count; ++i)
772 {
773 var property = typeof(TClass).GetProperty(schema[i].ColumnName, BindingFlags.IgnoreCase | BindingFlags.Public | BindingFlags.Instance);
774 if (property == null)
775 {
776 continue;
777 }
778
779 var value = Convert.ChangeType(Convert.ChangeType(reader.GetValue(i), reader.GetFieldType(i)), property.PropertyType);
780 property.SetValue(obj, value, null);
781 }
782
783 var item = (TClass)obj;
784 if (item == null)
785 {
786 return null;
787 }
788
789 result.Add(item);
790 }
791 }
792
793 return result;
794 }

808 public async Task<Dictionary<string, List<Dictionary<string, object>>>> getSqlRedShiftMultipleAsync(IEnumerable<(string, string)> namedQueries)
809 {
810 const string queryKeyColumn = "amm_query_key";
811
812 using var conn = new NpgsqlConnection(_rsConnectionString);
813 conn.Open();
814 if (conn.State != ConnectionState.Open)
815 {
816 return null;
817 }
818
819 var queries = new List<string>();
820 foreach (var (key, query) in namedQueries)
821 {
822#pragma warning disable SYSLIB1045 // Convert to 'GeneratedRegexAttribute'.
823 queries.Add(Regex.Replace(query, @"^((?:\r\n|\r|\n)*?\w*?SELECT\s+?(?:TOP\s+?\d+\s+?)?)", $@"$1'{key}' AS {queryKeyColumn}, ", RegexOptions.IgnoreCase));
824#pragma warning restore SYSLIB1045 // Convert to 'GeneratedRegexAttribute'.
825 }
826
827 var sql = string.Join(";", queries);
828 var cmd = new NpgsqlCommand(sql, conn);
829 var result = new Dictionary<string, List<Dictionary<string, object>>>();
830 var reader = await cmd.ExecuteReaderAsync();
831 do
832 {
833 while (await reader.ReadAsync())
834 {
835 var values = new object[reader.FieldCount];
836 var columns = reader.GetColumnSchema();
837 _ = reader.GetValues(values);
838
839 var key = (string)values[0];
840 result[key] = result.GetValueOrDefault(key) ?? [];
841 result[key].Add([]);
842 for (var i = 1; i < values.Length; ++i)
843 {
844 result[key].Last()[columns[i].ColumnName] = values[i];
845 }
846 }
847 } while (await reader.NextResultAsync());
848
849 return result;
