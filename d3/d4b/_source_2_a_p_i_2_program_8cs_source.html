126 protected static void ConfigureServices(IServiceCollection services, IConfigurationRoot config)

128 if (config.GetValue("JWTSettings:UseOwnToken", true))
129 {
130 _ = services.AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
131 .AddJwtBearer(options =>
132 {
133 options.TokenValidationParameters = new TokenValidationParameters
134 {
135 IssuerSigningKey = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(config["JWTSettings:SecretKey"])),
136 ValidateIssuerSigningKey = true,
137 ValidateIssuer = false,
138 ValidateAudience = false,
139 RequireExpirationTime = true,
140 ValidateLifetime = true,
141 ClockSkew = TimeSpan.Zero
142 };
143#pragma warning disable CS1998 // Async method lacks 'await' operators and will run synchronously
144 options.Events = new JwtBearerEvents()
145 {
146 OnTokenValidated = async ctx =>
147 {
148 var userIdClaim = ctx.Principal.Claims.FirstOrDefault(c => c.Type.Equals("userid", StringComparison.CurrentCultureIgnoreCase));
149 var repCodeClaim = ctx.Principal.Claims.FirstOrDefault(c => c.Type.Equals("repcode", StringComparison.CurrentCultureIgnoreCase));
150 var clientNameClaim = ctx.Principal.Claims.FirstOrDefault(c => c.Type.Equals("clientname", StringComparison.CurrentCultureIgnoreCase));
151
152 if (userIdClaim == null)
153 {
154 ctx.Fail("userId missing");
155 return;
156 }
157 else if (repCodeClaim == null)
158 {
159 ctx.Fail("repCode missing");
160 return;
161 }
162 else if (clientNameClaim == null)
163 {
164 ctx.Fail("clientName missing");
165 return;
166 }
167
168 var clientDbContext = ctx.HttpContext.RequestServices.GetRequiredService<ClientDbContext>();
169 var client = clientDbContext.Clients.FirstOrDefault(c => c.Name == clientNameClaim.Value);
170 if (client == null)
171 {
172 ctx.Fail("Invalid client");
173 return;
174 }
175
176 var appDbContext = ctx.HttpContext.RequestServices.GetRequiredService<ApplicationDbContext>();
177 var user = appDbContext.Users.FirstOrDefault(u => u.Id == userIdClaim.Value && u.RepCode == repCodeClaim.Value && u.ClientId == client.Id);
178 if (user == null)
179 {
180 ctx.Fail("Invalid user");
181 return;
182 }
183
184 var sessionService = ctx.HttpContext.RequestServices.GetRequiredService<ISessionService>();
185 sessionService.SetUserId(user.Id);
186 if (sessionService.Session.Id == null)
187 {
188 var mapperService = ctx.HttpContext.RequestServices.GetRequiredService<IMapper>();
189 sessionService.SetUserId(userIdClaim.Value);
190 if (!sessionService.SetSessionVariables(mapperService.Map<ClientVM>(client), user))
191 {
192 ctx.Fail("Error setting session variables");
193 return;
194 }
195 }
196 }
197 };
198#pragma warning restore CS1998 // Async method lacks 'await' operators and will run synchronously
199 });
200 }

202 {
