55 public async Task<ServiceResult<TemplateVM>> GetSingleByTemplateId(string templateId)
56 {
57 var templateResult = await Single(t => t.TemplateId == templateId, false, ["TemplateFields.Options", "TemplateThumbnailModels"]);
58 var template = templateResult.Data;
59
60 if (!string.IsNullOrWhiteSpace(template?.BackCanvasJson))
61 {
62 var templateBackCanvasJson = JsonHelper.Parse(template.BackCanvasJson);
63 if (templateBackCanvasJson["background"] != null && templateBackCanvasJson["background"]["source"] != null)
64 {
65 templateBackCanvasJson["background"]["source"] = configuration.ReplaceStageFolder(template.BackImage);
66
67 template.BackCanvasJson = templateBackCanvasJson.ToString();
68 }
69 }
70
71 if (!string.IsNullOrWhiteSpace(template?.FrontCanvasJson))
72 {
73 var templateFrontCanvasJson = JsonHelper.Parse(template.FrontCanvasJson);
74 if (templateFrontCanvasJson["background"] != null && templateFrontCanvasJson["background"]["source"] != null)
75 {
76 templateFrontCanvasJson["background"]["source"] = configuration.ReplaceStageFolder(template.FrontImage);
77
78 template.FrontCanvasJson = templateFrontCanvasJson.ToString();
79 }
80 }
81
82 if (templateResult.ErrorMessage != "")
83 {
84 logger.LogInformation("{ErrorMessage}", templateResult.ErrorMessage);
85 return new ServiceResult<TemplateVM>(null, templateResult.ErrorMessage);
86 }
87
88 return new ServiceResult<TemplateVM>(template, "");

100 case "center":
101 getSignatureBlockResult = await AmazonS3Service.DownloadHTMLFromContent("simpleSignatureBlock_Center.html", "SignatureBlock");
102 break;
103 case "right":
104 getSignatureBlockResult = await AmazonS3Service.DownloadHTMLFromContent("simpleSignatureBlock_Right.html", "SignatureBlock");
105 break;
106 default:
107 return new ServiceResult<string>(null, "SignatureBlock not found!");
108 }
109
110 if (!string.IsNullOrWhiteSpace(getSignatureBlockResult.ErrorMessage))
111 {
112 return new ServiceResult<string>(null, getSignatureBlockResult.ErrorMessage);
