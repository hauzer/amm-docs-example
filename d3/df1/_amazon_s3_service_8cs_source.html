108 private async Task<ServiceResult<AmazonResponse>> DownloadStringFromUrl(string url, string s3Bucket = null)
109 {
110 if (string.IsNullOrEmpty(s3Bucket))
111 {
112 s3Bucket = Bucket;
113 }
114
115 var downloadTemplateRequest = new GetObjectRequest()
116 {
117 BucketName = s3Bucket,
118 Key = url
119 };
120 var response = await _amazonClient.GetObjectAsync(downloadTemplateRequest);
121 var sr = new StreamReader(response.ResponseStream, Encoding.UTF8);
122
123 var amazonResponse = new AmazonResponse { Successful = true, Message = sr.ReadToEnd() };
124 return new ServiceResult<AmazonResponse>(amazonResponse, "");

175 public async Task<ServiceResult<AmazonResponse>> UploadCSV(IFormFile content, string fileName, string bucket)
176 {
177 if (content == null)
178 {
179 return new ServiceResult<AmazonResponse>(null, "CSV file not sent!");
180 }
181
182 var ms = new MemoryStream();
183 content.CopyTo(ms);
184 var uploadCSVRequest = new PutObjectRequest()
185 {
186 BucketName = bucket,
187 Key = fileName.Replace("\\","/"),
188 ContentType = "text/csv",
189 InputStream = ms,
190 //CannedACL = S3CannedACL.PublicRead
191 };
192 _ = ms.Seek(0, SeekOrigin.Begin);
193
194 _ = await _amazonClient.PutObjectAsync(uploadCSVRequest);
195 ms.Close();
196
197 return new ServiceResult<AmazonResponse>(new AmazonResponse { Successful = true, Path = _s3UrlPrefix + "/" + bucket + "/" + fileName }, "");
198 }
