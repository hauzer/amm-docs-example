294 public async Task<ServiceResult<PMSPersonaDTO>> GetPersonaInstance(string personaInstanceId, string userId)
295 {
296 var getPersonaResult = await Get<GetPersonaInstanceResponse>("trimmed/personainstances/" + personaInstanceId + "/" + userId, _serviceName);
297 if (getPersonaResult == null)
298 {
299 return new ServiceResult<PMSPersonaDTO>(null, $"GetTrimmedPersonaInstance({personaInstanceId}, {userId}) Response error.");
300 }
301 else if (getPersonaResult.data == null || string.IsNullOrWhiteSpace(getPersonaResult.id))
302 {
303 return new ServiceResult<PMSPersonaDTO>(null, $"PersonaInstance {personaInstanceId} not found.");
304 }
305 else
306 {
307 var result = new PMSPersonaDTO
308 {
309 Active = getPersonaResult.isActive,
310 Id = getPersonaResult.id,
311 PersonaInstanceName = getPersonaResult.name
312 };
313 foreach (var infoItem in getPersonaResult.data)
314 {
315 result.Data.Add(infoItem.tag, infoItem.value);
316 }
317
318 SetFullPaths(result);
319 if (!result.Data.ContainsKey("PersonaGUID"))
320 {
321 result.Data.Add("PersonaGUID", getPersonaResult.id);
322 }
323
324 return new ServiceResult<PMSPersonaDTO>(result, "");
325 }

407 private void SetFullPaths(PMSPersonaDTO personalInfo)
408 {
409 var bucket = _configuration["AmazonS3Settings:S3ImagesBucket"];
410 var folder = _configuration["AmazonS3Settings:S3ImagesFolder"];
411
412 personalInfo.Data["ARInsuranceNumber"] = personalInfo.Data.GetValueOrDefault("State") == "AR" ? personalInfo.Data.GetValueOrDefault("InsuranceNumber") : "";
413 personalInfo.Data["CAInsuranceNumber"] = personalInfo.Data.GetValueOrDefault("State") == "CA" ? personalInfo.Data.GetValueOrDefault("InsuranceNumber") : "";
414
415 var imagesKeys = new List<string>
416 {
417 "ProfilePhoto",
418 "CompanyLogo",
419 "SignatureImage",
420 "MNMImage",
421 "NNM1LineLinearImage",
422 "NNM3BoxMEWAImage",
423 "NNMStackedVer1Image",
424 "NNMStackedVer2Image",
425 "NNMStackedVer3Image",
426 "NNMStackedVer4Image",
427 "GP_GroupPhoto",
428 "GP_SignatureImage_1",
429 "GP_SignatureImage_2",
430 "GP_SignatureImage_3"
431 };
432
433 foreach (var key in imagesKeys)
434 {
435 var thumbnailKey = key + "Thumbnail";
436 var filename = personalInfo.Data.GetValueOrDefault(key);
437
438 personalInfo.Data[key] = string.IsNullOrEmpty(filename) ? "" : ("https://" + bucket + ".s3-us-west-2.amazonaws.com/" + folder + "/" + filename);
439
440 var extension = Path.GetExtension(filename);
441 if (extension is ".pdf" or ".eps")
442 {
443 extension = ".png";
444 }
445
446 personalInfo.Data[thumbnailKey] = string.IsNullOrEmpty(filename) ? "" : ("https://" + bucket + ".s3-us-west-2.amazonaws.com/" + folder + "/" + Path.GetFileNameWithoutExtension(filename) + "_thumbnail" + extension);
447 }
448 }
