187 public override ViewModelValidation Validate(string clientName)
188 {
189 var result = base.Validate(clientName);
190
191 var normalizedState = (State is "NONE" or "0") ? "" : State;
192 var physicalAddress = new List<string>(new string[] { Address1, normalizedState, City, Zip });
193
194 var hasValidEmail = !string.IsNullOrEmpty(Email) && !string.IsNullOrWhiteSpace(Email);
195 var hasValidPhysicalAddress =
196 physicalAddress.All(x => !string.IsNullOrEmpty(x) && !string.IsNullOrWhiteSpace(x)) ||
197 physicalAddress.All(x => string.IsNullOrEmpty(x) || string.IsNullOrWhiteSpace(x));
198
199 if ((!hasValidEmail && !hasValidPhysicalAddress) || !hasValidPhysicalAddress)
200 {
201 result.Success = false;
202 result.Errors.Add(new KeyValuePair<string, string>(
203 "Email/Address1/State/City/Zip",
204 "A contact must at least have either an Email OR all of the following: Address1, State, City, and Zip."
205 ));
206 }
207
208 if (clientName == "RJ")
209 {
210 if (string.IsNullOrEmpty(Address1) || string.IsNullOrWhiteSpace(Address1))
211 {
212 result.Success = false;
213 result.Errors.Add(new KeyValuePair<string, string>("Address1", "Required"));
214 }
215
216 if (string.IsNullOrEmpty(State) || string.IsNullOrWhiteSpace(State))
217 {
218 result.Success = false;
219 result.Errors.Add(new KeyValuePair<string, string>("State", "Required"));
220 }
221
222 if (string.IsNullOrEmpty(City) || string.IsNullOrWhiteSpace(City))
223 {
224 result.Success = false;
225 result.Errors.Add(new KeyValuePair<string, string>("City", "Required"));
226 }
227
228 if (string.IsNullOrEmpty(Zip) || string.IsNullOrWhiteSpace(Zip))
229 {
230 result.Success = false;
231 result.Errors.Add(new KeyValuePair<string, string>("Zip", "Required"));
232 }
233
234 if (string.IsNullOrEmpty(Title) || string.IsNullOrWhiteSpace(Title))
235 {
236 result.Success = false;
237 result.Errors.Add(new KeyValuePair<string, string>("Title", "Required"));
238 }
239
240 if (string.IsNullOrEmpty(MFGroupName) || string.IsNullOrWhiteSpace(MFGroupName))
241 {
242 result.Success = false;
