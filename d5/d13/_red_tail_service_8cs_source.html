82 public async Task<ServiceResult<List<SelectListItem>>> PullRedtailTagGroups(string username = "", string password = "")
83 {
84 if (string.IsNullOrWhiteSpace(username) || string.IsNullOrWhiteSpace(password))
85 {
86 var user = _sessionService.SessionGetUser();
87 username = user.RedtailUsername;
88 password = user.RedtailPassword;
89 }
90
91 var redtailAuth = await Get<List<RedTailVM>>("/crm/v1/rest/settings/taggroups", username, password);
92 if (redtailAuth == null)
93 {
94 return new ServiceResult<List<SelectListItem>>(null, "Credentials are not valid.");
95 }
96
97 var listOfSelectedItems = new List<SelectListItem>();
98 foreach (var item in redtailAuth)
99 {
100 var newListItem = new SelectListItem
101 {
102 Text = item.Name,
103 Value = item.RecID.ToString()
104 };
105 listOfSelectedItems.Add(newListItem);
106 };
107 listOfSelectedItems = listOfSelectedItems.OrderBy(x => x.Text).ToList();
108
109 return new ServiceResult<List<SelectListItem>>(listOfSelectedItems, "");
110 }
