62 public Paginate(IEnumerable<T> source, PaginateConfiguration paginateConfiguration)
63 {
64 var enumerable = source as T[] ?? source.ToArray();
65
66 if (paginateConfiguration is null)
67 {
68 throw new Exception("Paginate configuration was not sent.");
69 }
70
71 var index = paginateConfiguration.Index < 0 ? 0 : paginateConfiguration.Index;
72 var from = paginateConfiguration.From < 0 ? 0 : paginateConfiguration.From;
73 var size = paginateConfiguration.Size < 1 ? source.Count() : paginateConfiguration.Size;
74
75 if (from > index)
76 {
77 throw new ArgumentException($"indexFrom: {from} > pageIndex: {index}, must indexFrom <= pageIndex");
78 }
79
80 Index = index;
81 Size = size;
82 From = from;
83 Count = source.Count();
84 Pages = Count > 1 ? (int)Math.Ceiling(Count / (double)Size) : 0;
