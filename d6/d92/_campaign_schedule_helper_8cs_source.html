17 public List<DateTime> GetScheduledDates(RecurrenceVM recurrence, DateTime start, bool futureDatesOnly = true)
18 {
19 result = [];
20 var maxOcurrence = 10;
21 if (recurrence.MaxOcurrence.HasValue)
22 {
23 maxOcurrence = recurrence.MaxOcurrence.Value < 1 ? 10 : recurrence.MaxOcurrence.Value;
24 }
25
26 scheduledDate = start;
27 occurence = 1;
28 while (occurence <= maxOcurrence)
29 {
30 if (recurrence.Frequency == Frequency.Minutely)
31 {
32 SetMinutelySchedule(recurrence.Interval);
33 }
34 else if (recurrence.Frequency == Frequency.Daily)
35 {
36 SetDailySchedule(recurrence.Interval);
37 }
38 else if (recurrence.Frequency == Frequency.Weekly)
39 {
40 SetWeeklyDate(recurrence.RecursUntil, recurrence.Interval, recurrence.DaysOfWeek.Split(",").ToList());
41 }
42 else if (CheckMonthlyOffset(recurrence)) // offset monthly
43 {
44 SetMonthlyDateOffset(start, recurrence.RecursUntil, recurrence.Interval, recurrence.WeekDaysType, recurrence.DaySequence);
45 }
46 else if (recurrence.Frequency == Frequency.Monthly)
47 {
48 SetMonthlyDate(start, recurrence.RecursUntil, recurrence.Interval, recurrence.Day);
49 }
50 else if (CheckYearlyOffset(recurrence)) //offset yearly
51 {
52 SetYearlyDateOffset(start, recurrence.RecursUntil, recurrence.DaySequence, recurrence.WeekDaysType, recurrence.Month);
53 }
54 else if (recurrence.Frequency == Frequency.Yearly)
55 {
56 SetYearlyDate(start, recurrence.RecursUntil, recurrence.Month, recurrence.Day);
57 }
58
59 occurence += 1;
60 if (recurrence.RecursUntil != null && scheduledDate > recurrence.RecursUntil)
61 {
62 occurence = maxOcurrence + 1;
63 }
64 }
65
66 if (futureDatesOnly)
67 {
68 result = result.Where(x => x > DateTime.Now).ToList();
69 }
70
71 return result.OrderBy(x => x).Take(maxOcurrence).ToList();

401 private static int FindDay(int year, int month, DayOfWeek day, int daySequenceNumber)
402 {
403 if (daySequenceNumber is < 1 or > 4)
404 {
405 return -1;
406 }
407
408 var fd = new DateTime(year, month, 1);
409 var daysNeeded = day - fd.DayOfWeek;
410 if (daysNeeded < 0)
411 {
412 daysNeeded += 7;
413 }
414
415 var resultDay = daysNeeded + 1 + (7 * (daySequenceNumber - 1));
416 if (resultDay > (fd.AddMonths(1) - fd).Days)
417 {
418 return -1;
419 }
420
421 return resultDay;
