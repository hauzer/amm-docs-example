16 public static string ExcludeImgFromString(this string htmlContent, string hashTagImage)
17 {
18 var indexOfHashTagImage = htmlContent.IndexOf(hashTagImage);
19 if (indexOfHashTagImage == -1)
20 {
21 return htmlContent;
22 }
23
24 var indexOfOpenBracket = -1;
25 var lengthForRemoving = 0;
26 for (var i = indexOfHashTagImage; i > 0; i--)
27 {
28 if (htmlContent[i] == '>')
29 {
30 return htmlContent;
31 }
32
33 if (htmlContent[i] == '<')
34 {
35 indexOfOpenBracket = i;
36 break;
37 }
38 }
39
40 for (var i = indexOfOpenBracket; i < htmlContent.Length; i++)
41 {
42 lengthForRemoving++;
43 if (htmlContent[i] == '>')
44 {
45 break;
46 }
47 }
48
49 return htmlContent.Replace(htmlContent.Substring(indexOfOpenBracket, lengthForRemoving), "");
50 }
