609 public async Task<ServiceResult<LoginResponseVM>> SSOSignIn(int clientId, string repId, string userId, bool admin)
610 {
611 var clientSingleResponse = await clientService.Single(clientId);

613 {
614 logger.LogInformation("Unknown client.");

616 }
617
618 var client = clientSingleResponse.Data;
619
620 var user = await dbContext.Users.AsNoTracking().Include(x => x.AccountStatus).FirstOrDefaultAsync(x => x.RepCode == repId && x.ClientId == clientId);

622 {
623 if (string.IsNullOrEmpty(user.Email))
624 {
625 const string message = "User was registered on AMM before having an email on DBC. Please use a different user.";

629
630 _ = _sessionService.SetSessionVariables(client, user);

632 new LogInVM

634 UserName = user.UserName,
635 RememberMe = false,
636 Client = client.Name,
637 Password = user.UserName.ToLower() + "_Connect101"

639 user

645 if (invalidUser != null)

647 _ = dbContext.Users.Remove(invalidUser);
648 _ = await dbContext.SaveChangesAsync();
649 }

651 // If this is a new user, register it

653 if (!string.IsNullOrEmpty(dbcServiceGetUserResult.ErrorMessage))

666 logger.LogError(message);
667 return new ServiceResult<LoginResponseVM>(null, message);
668 }
669
670 _sessionService.Ephemeral = true; // Use an in-memory session before we create the user in the DB.
671 _ = _sessionService.SetSessionVariables(client, user);
672
673 // TODO: Do we still need this?

678 }
679

682 if (!string.IsNullOrEmpty(registerAdvisorResponse.ErrorMessage))

684 return new ServiceResult<LoginResponseVM>(null, registerAdvisorResponse.ErrorMessage);

688 user = registerAdvisorResponse.Data;
689 return await SignIn(
690 new LogInVM
691 {
692 UserName = user.UserName,
693 RememberMe = false,
694 Client = client.Name,
695 Password = user.UserName.ToLower() + "_Connect101"
