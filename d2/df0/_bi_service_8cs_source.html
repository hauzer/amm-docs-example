56 public async Task<DTOCamapigns> GetCampaignsWithElements(AdvisorFilter filter)
57 {
58 DTOCamapigns result;
59 var userCampaignsResult = await _userCampaignService.GetList();
60 if (!string.IsNullOrEmpty(userCampaignsResult.ErrorMessage))
61 {
62 Log.Error(userCampaignsResult.ErrorMessage);
63 return [];
64 }
65
66 var programIds = userCampaignsResult.Data?.Where(uc => uc.ProgramId != 0 && uc.UserId == _sessionService.SessionGetUserId()).Select(uc => (long)uc.ProgramId);
67 if (programIds is null || !programIds.Any())
68 {
69 return [];
70 }
71
72 result = await _redShiftService.getCampaignsAsync(filter.AccountId, filter.CampaingIds, programIds, filter.ThisMonth);
73 if (result != null)
74 {
75 // get Campaign elements
76 var campaignIds = new List<long>();
77 foreach (var campaign in result)
78 {
79 campaignIds.Add(campaign.Id);
80 }
81
82 if (campaignIds.Count > 0)
83 {
84 var elements = await _redShiftService.getCamapignElements(filter.AccountId, campaignIds);
85 foreach (var element in elements)
86 {
87 var campaign = result.findCampaign(element.CampaignId);
88 if (campaign != null)
89 {
90 if (element.Service == "Email")
91 {
92 campaign.Elements.Email = element.Count;
93 }
94 else if (element.Service == "Microsite")
95 {
96 campaign.Elements.Microsite = element.Count;
97 }
98 else if (element.Service == "DirectMail")
99 {
100 campaign.Elements.DirectMail = element.Count;
101 }
102 else if (element.Service == "Cloud Connect")
103 {
104 campaign.Elements.Social = element.Count;
105 }
106 else
107 {
108 campaign.Elements.Rest += element.Count;
109 }
110 }
111 }
112 }
113
114 return result;
115 }
116
117 return [];
118 }

483 private static SegmentationFilter getMFAccountIdFilter(IEnumerable<string> accountIds)
484 {
485 var sb = new StringBuilder();
486 _ = sb.Append("<Filter CriteriaJoinOperator=\"|\">");
487 var row = 1;
488 var result = new SegmentationFilter
489 {
490 CriteriaJoinOperator = "|"
491 };
492 foreach (var accountId in accountIds)
493 {
494 var criteria = new SegmentationFilterCriteria("accountid", "Equal", accountId, row);
495 result.Criterias.Add(criteria);
496 row++;
497 }
498
499 return result;
500 }
