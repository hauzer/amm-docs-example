91 public async Task<ServiceResult<string>> Sharing(SharingRequest sharingRequest)
92 {
93 if (sharingRequest.PublishForUser?.Accounts?.Count < 1)
94 {
95 _logger.LogInformation("There isn't any provider.");
96 return new ServiceResult<string>(null, "There isn't any provider.");
97 }
98
99 if (sharingRequest.SocialContent is null)
100 {
101 _logger.LogInformation("No parts given for posting.");
102 return new ServiceResult<string>(null, "No parts given for posting.");
103 }
104
105 if (string.IsNullOrWhiteSpace(sharingRequest.PublishForUser?.SocialToken))
106 {
107 _logger.LogInformation("No user token given.");
108 return new ServiceResult<string>(null, "No user token given.");
109 }
110
111 var checkMsgRes = await CheckAndDownloadMessage(sharingRequest.SocialContent.Text);
112 if (!string.IsNullOrWhiteSpace(checkMsgRes.ErrorMessage))
113 {
114 _logger.LogInformation("{ErrorMessage}", checkMsgRes.ErrorMessage);
115 return new ServiceResult<string>(null, checkMsgRes.ErrorMessage);
116 }
117
118 sharingRequest.SocialContent.Text = checkMsgRes.Data;
119
120 string response = null;
121 foreach (var account in sharingRequest.PublishForUser?.Accounts)
122 {
123 var line = new SocialPostLogLine();
124 switch (account.SocialProvider)
125 {
126 case SocialProvider.Linkedin:
127 line.Network = "Linkedin";
128 line.SharingRequest = JsonHelper.Serialize(sharingRequest);
129 var proceed = await CheckRefreshTokenForWarning(account, sharingRequest.PublishForUser?.SocialToken);
130 if (proceed)
131 {
132 var linkedinPublishResult = await _linkedinService.SharePost(sharingRequest.SocialContent, account, sharingRequest.PublishForUser?.SocialToken);
133 line.SharingResponse = JsonHelper.Serialize(linkedinPublishResult.Data);
134 line.CampaignElementId = sharingRequest.PublishForUser.CampaignElementId;
135 if (!string.IsNullOrWhiteSpace(linkedinPublishResult.ErrorMessage))
136 {
137 _logger.LogInformation("{ErrorMessage}", linkedinPublishResult.ErrorMessage);
138 line.SharingError = linkedinPublishResult.ErrorMessage;
139 }
140 }
141
142 break;
143 case SocialProvider.Twitter:
144 line.Network = "Twitter";
145 line.SharingRequest = JsonHelper.Serialize(sharingRequest);
146 var publishRes = await _twitterService.PublishTweet(sharingRequest.SocialContent, account, sharingRequest.PublishForUser.SocialToken);
147 response = line.SharingResponse = publishRes.Data;
148 line.CampaignElementId = sharingRequest.PublishForUser.CampaignElementId;
149 if (publishRes.ErrorMessage != "")
150 {
151 _logger.LogInformation("{ErrorMessage}", publishRes.ErrorMessage);
152 line.SharingError = publishRes.ErrorMessage;
153 }
154
155 break;
156 default:
157 break;
158 }
159
160 line.SharingResponse ??= "";
161 _ = _dbContext.LogSocialPost.Update(line);
162 _ = await _dbContext.SaveChangesAsync();
163 }
164
165 return new ServiceResult<string>(response, "");
166 }

168 private async Task<bool> CheckRefreshTokenForWarning(SocialAccount account, string socialToken)
169 {
170 if (string.IsNullOrWhiteSpace(socialToken))
171 {
172 return false;
173 }
174
175 var proceed = true;
176 var getLinkedinAccountResult = await _dbContext.LinkedinAccounts.FirstOrDefaultAsync(predicate: x => x.Enabled && x.SocialToken == socialToken && x.Id == account.AccountId);
177 if (getLinkedinAccountResult == null)
178 {
179 getLinkedinAccountResult = await _dbContext.LinkedinAccounts.FirstOrDefaultAsync(predicate: x => x.Enabled && x.SocialToken == socialToken);
180 if (getLinkedinAccountResult == null)
181 {
182 return false;
183 }
184 }
185 // check refreshToken status
186 var checkTokensStatusForWarningResponse = await _linkedinService.CheckTokensStatusForWarning(getLinkedinAccountResult.AccessToken, getLinkedinAccountResult.RefreshToken, getLinkedinAccountResult.ClientId ?? -1);
187
188 if (checkTokensStatusForWarningResponse != null && (checkTokensStatusForWarningResponse.Data.WarningStatus == 1 || checkTokensStatusForWarningResponse.Data.WarningStatus == 2))
189 {
190 if (checkTokensStatusForWarningResponse.Data.WarningStatus == 2)
191 {
192 proceed = false;
193 }
194 // preparing to send email
195 var groupName = "EmailSettings";
196 var settings = GetSettings(groupName, getLinkedinAccountResult.ClientId ?? -1);
197 if (settings.Count == 0)
198 {
199 return proceed;
200 }
201
202 List<PMSPersonaDTO> personas;
203 // get list of PersonaInstances with provided socialToken for warning email
204 var getFilteredPersonasResponse = await _personaMicroservice.GetFilteredPersonas("SocialToken", socialToken);
205 if (getFilteredPersonasResponse.Data != null)
206 {
207 personas = getFilteredPersonasResponse.Data;
208 if (personas.Count != 0)
209 {
210 var preparingEmailResponse = await PreparingEmail(settings, getLinkedinAccountResult.ClientId ?? -1, checkTokensStatusForWarningResponse.Data);
211 if (preparingEmailResponse != null && !string.IsNullOrEmpty(preparingEmailResponse.TemplateUrl))
212 {
213 foreach (var persona in personas)
214 {
215 _ = await _sendEmailService.SendEmail(
216 preparingEmailResponse.FromEmail,
217 preparingEmailResponse.FromName,
218 preparingEmailResponse.Subject,
219 preparingEmailResponse.TemplateUrl,
220 persona.Data,
221 preparingEmailResponse.S3Bucket,
222 preparingEmailResponse.RefreshTokenLastingDays
223 );
224 }
225 }
226 }
227 }
228 else
229 {
230 if (!string.IsNullOrWhiteSpace(getFilteredPersonasResponse.ErrorMessage))
231 {
232 _logger.LogError("Failed to retreive PersonaInstances with selected SocialToken. {ErrorMessage}", getFilteredPersonasResponse.ErrorMessage);
233 }
234 else
235 {
236 _logger.LogError("Failed to retreive PersonaInstances with selected SocialToken.");
237 }
238 }

300 public bool Authenticate(IHeaderDictionary headerDictionary)
301 {
302 var mfAuthToken = headerDictionary.Authorization.ToString();
303 var appAuthToken = _configuration["MindfireSettings:AppAuthToken"].Replace("::", ":");
304 if (string.IsNullOrWhiteSpace(mfAuthToken))
305 {
306 _logger.LogInformation("Share Authorization failed!");
307 return false;
308 }
309
310 var decodedAuthToken = System.Text.Encoding.UTF8.GetString(Convert.FromBase64String(mfAuthToken["Basic ".Length..]));
311 _logger.LogInformation("Decoded Auth: {Token}", decodedAuthToken);
312 if (decodedAuthToken != appAuthToken)
313 {
314 _logger.LogInformation("Share Authorization failed!");
315 return false;
316 }
317
318 _logger.LogInformation("Share Authorization passed!");
319 return true;
320 }

349 var activeList = new List<UserProvider>();
350 if (string.IsNullOrWhiteSpace(checkSocialVM.SocialToken))

352 foreach (var provider in checkSocialVM.Providers)
353 {
354 UnlinkedProvider(checkSocialVM.AllActive, provider.Provider.ToString(), provider, activeList);
355 }
356
357 return new ServiceResult<CheckSocialVM>(checkSocialVM, "");
358 }
359
360 foreach (var provider in checkSocialVM.Providers)
361 {
362 switch (provider.Provider)
363 {
364 case SocialProvider.Linkedin:
365 if (await _linkedinService.DoesExist(predicate: x => x.SocialToken == checkSocialVM.SocialToken))
366 {
367 var removedAccountCount = 0;
368 var serviceResult = await _linkedinService.GetBySocialToken(checkSocialVM.SocialToken);
369 if (!string.IsNullOrWhiteSpace(serviceResult.ErrorMessage) || serviceResult.Data == null)
370 {
371 UnlinkedProvider(checkSocialVM.AllActive, "Linkedin", provider, activeList);
372 continue;
373 }
374
375 foreach (var item in serviceResult.Data)
376 {
377 var profileRes = await _linkedinService.GetProfile(item.AccessToken, item.ClientId ?? -1);
378 if (profileRes.Data == null)
379 {
380 _ = await _linkedinService.Remove(new UnlinkSocialVM()
381 {
382 DbId = item.Id,
383 SocialProvider = SocialProvider.Linkedin
384 });
385 removedAccountCount++;
386 continue;
387 }
388
389 var warningStsatus = 2;
390 var tokenStatusError = "";
391
392 var tokenWarningResponse = await _linkedinService.CheckTokensStatusForWarning(item.AccessToken, item.RefreshToken, item.ClientId ?? -1);
393 if (tokenWarningResponse.Data == null)
394 {
395 if (!string.IsNullOrEmpty(tokenWarningResponse.ErrorMessage))
396 {
397 warningStsatus = 31;
398 tokenStatusError = tokenWarningResponse.ErrorMessage;
399 }
400 else
401 {
402 warningStsatus = 32;
403 tokenStatusError = "warningStatusResponse error";
404 }
405 }
406
407 activeList.Add(new UserProvider()
408 {
409 SocialName = "Linkedin",
410 FirstName = profileRes.Data.LocalizedFirstName,
411 LastName = profileRes.Data.LocalizedLastName,
412 IsLinked = true,
413 DbId = item.Id,
414 AccountId = profileRes.Data.Id,
415 Provider = SocialProvider.Linkedin,
416 SocialURL = !string.IsNullOrEmpty(profileRes.Data.VanityName) ? "https://www.linkedin.com/in/" + profileRes.Data.VanityName + "/detail/recent-activity/shares/" : "",
417 TokenWarningStatus = tokenWarningResponse.Data != null ? tokenWarningResponse.Data.WarningStatus : warningStsatus,
418 TokenStatusErrorMessage = tokenStatusError,
419 RefreshTokenLastingDays = tokenWarningResponse.Data != null ? tokenWarningResponse.Data.RefreshTokenLastingDays : 0
420 });
421 }
422
423 //if all accounts that were linked for Profile/Persona have invalid access token then remove them and return Linkedin isLinked=false.
424 if (removedAccountCount == serviceResult.Data.Count)
425 {
426 UnlinkedProvider(checkSocialVM.AllActive, "Linkedin", provider, activeList);
427 }
428 }
429 else
430 {
431 UnlinkedProvider(checkSocialVM.AllActive, "Linkedin", provider, activeList);
432 }
433
434 break;
435 case SocialProvider.Twitter:
436 if (await _twitterService.DoesExist(predicate: x => x.SocialToken == checkSocialVM.SocialToken))
437 {
438 var serviceResult = await _twitterService.GetBySocialToken(checkSocialVM.SocialToken);
439 if (!string.IsNullOrWhiteSpace(serviceResult.ErrorMessage) || serviceResult.Data == null)
440 {
441 UnlinkedProvider(checkSocialVM.AllActive, "Twitter", provider, activeList);

443 }
444

446 foreach (var item in serviceResult.Data)
447 {
448 var profileRes = _twitterService.GetProfile(item);
449 if (profileRes.Data == null)
450 {
451 _ = await _twitterService.Remove(new UnlinkSocialVM()
452 {
453 DbId = item.Id,
454 SocialProvider = SocialProvider.Twitter

458 }
459
460 activeList.Add(new UserProvider()

462 SocialName = "Twitter",

464 LastName = "@" + profileRes.Data.ScreenName,
465 IsLinked = true,
466 DbId = item.Id,

470 });
471 }
472
473 //if all accounts that were linked for Profile/Persona have invalid access token then remove them and return Twitter isLinked=false.
474 if (removedAccountCount == serviceResult.Data.Count)
475 {
476 UnlinkedProvider(checkSocialVM.AllActive, "Twitter", provider, activeList);
477 }
478 }
479 else
480 {
481 UnlinkedProvider(checkSocialVM.AllActive, "Twitter", provider, activeList);

485 }
486 }
487
488 checkSocialVM.Providers = activeList;
489 return new ServiceResult<CheckSocialVM>(checkSocialVM, "");
