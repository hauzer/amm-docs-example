53 public async Task<ServiceResult<SettingsVM>> AddOrUpdate(SettingsVM settingsVM)
54 {
55 var settingsGroup = settingsVM.SettingsGroupId;
56
57 var settings = mapper.Map<Settings>(settingsVM);
58 var model = await dbContext.Settings.AsNoTracking().FirstOrDefaultAsync(x => x.Name == settings.Name && x.SettingsGroupId == settings.SettingsGroupId && x.Id != settings.Id);
59 if (model != null)
60 {
61 // If model.Enabled == true return error message
62 // If model.Enabled == false update existing data with new values, keeping original Id
63 if (model.Enabled)
64 {
65 logger.LogInformation("Settings was not updated");
66 return new ServiceResult<SettingsVM>(null, "Settings was not updated");
67 }
68 else
69 {
70 settings.Enabled = true;
71 settings.Id = model.Id;
72 _ = dbContext.Settings.Update(settings);
73 }
74 }
75 else if (settings.Id == 0)
76 {
77 settings.Enabled = true;
78 _ = await dbContext.Settings.AddAsync(settings);
79 }
80 else
81 {
82 _ = dbContext.Settings.Update(settings);
83 }
84
85 _ = dbContext.SaveChanges();
86 var returnModel = await Single(settings.Id);
87 if (!string.IsNullOrWhiteSpace(returnModel.ErrorMessage))
88 {
89 logger.LogInformation("{ErrorMessage}", returnModel.ErrorMessage);
90 return new ServiceResult<SettingsVM>(null, returnModel.ErrorMessage);
91 }
92
93 return new ServiceResult<SettingsVM>(returnModel.Data, "");
94 }

96 public ServiceResult<SettingsVM> SingleByName(string name, int? clientId)
97 {
98 var settings = dbContext.Settings.AsNoTracking().FirstOrDefault(sg => sg.Enabled && sg.Name == name && sg.ClientId == clientId);

101 logger.LogError("Requested settings do not exist.");
102 return new ServiceResult<SettingsVM>(null, "Requested settings do not exist.");
103 }
104
105 var settingsVM = mapper.Map<SettingsVM>(settings);
