120 public virtual async Task<string> UpdateSocialURLs(PMSPersonaDTO personalInfo)
121 {
122 if (string.IsNullOrEmpty(personalInfo.Data.GetValueOrDefault("SocialToken")))
123 {
124 return "";
125 }
126
127 var checkProfileRes = await _socialNetworkService.CheckSocialForUser(new CheckSocialVM()
128 {
129 AllActive = true,
130 SocialToken = personalInfo.Data.GetValueOrDefault("SocialToken"),
131 Providers = [
132 new() { Provider = SocialProvider.Linkedin },
133 new() { Provider = SocialProvider.Twitter }
134 ]
135 });
136 foreach (var provider in checkProfileRes.Data.Providers)
137 {
138 _ = provider.SocialName;
139 switch (provider.Provider)
140 {
141 case SocialProvider.Linkedin:
142 if (!string.IsNullOrEmpty(provider.SocialURL))
143 {
144 if (string.IsNullOrEmpty(personalInfo.LinkedInURL))
145 {
146 personalInfo.LinkedInURL = provider.SocialURL;
147 }
148 }
149
150 break;
151 case SocialProvider.Twitter:
152 if (!string.IsNullOrEmpty(provider.SocialURL))
153 {
154 if (string.IsNullOrEmpty(personalInfo.TwitterURL))
155 {
156 personalInfo.TwitterURL = provider.SocialURL;
157 }
158 }
159
160 break;
161 }
162 }
163
164 return "";
165 }

329 public virtual async Task<ServiceResult<List<TModelVM>>> GetList(Expression<Func<TModel, bool>> predicate = null,
330 Expression<Func<TModel, object>> orderBy = null,
331 bool descending = false,
332 bool disableTracking = false,
333 string[] navigationPaths = null,
334 bool clientSide = false,
335 params Expression<Func<TModel, object>>[] include)
336 {
337 List<TModel> models;
338 var cId = _sessionService.SessionGetClientId();
339 var query = _dbSet.Where(x => x.Enabled && ((x.ClientId == 0) || (x.ClientId == cId)));
340 if (disableTracking)
341 {
342 query = query.AsNoTracking();
343 }
344
345 if (include?.Length > 0)
346 {
347 foreach (var includeItem in include)
348 {
349 query = query.Include(includeItem);
350 }
351 }
352
353 if (navigationPaths?.Length > 0)
354 {
355 foreach (var navigationPath in navigationPaths)
356 {
357 query = query.Include(navigationPath);
358 }
359 }
360
361 if (clientSide)
362 {
363 query = query.AsEnumerable().AsQueryable();
364 }
365
366 if (predicate != null)
367 {
368 query = query.Where(predicate);
369 }
370
371 if (orderBy != null)
372 {
373 if (descending)
374 {
375 models = await query.OrderByDescending(orderBy).ToListAsync();
376 }
377 else
378 {
379 models = await query.OrderBy(orderBy).ToListAsync();
380 }
381 }
382 else
383 {
384 models = await query.ToListAsync();
385 }
386
387 return new ServiceResult<List<TModelVM>>(mapper.Map<List<TModelVM>>(models), "");
