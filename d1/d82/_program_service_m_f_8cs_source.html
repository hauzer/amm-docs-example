357 public async Task<ServiceResult<ProgramCheckinResponse>> StopCampaign(int programId, string ticket, string comment)
358 {
359 if (programId == 0)
360 {
361 return new ServiceResult<ProgramCheckinResponse>(null, "Campaign not launched.");
362 }
363
364 var credentials = new RequestCredentials(configuration)
365 {
366 Ticket = ticket
367 };
368
369 var checkOutResponse = await CheckOutProgram(credentials, programId);
370 if (checkOutResponse.ErrorMessage != "")
371 {
372 _ = await UndoProgramChanges(ticket, programId);
373 checkOutResponse = await CheckOutProgram(credentials, programId);
374 if (!string.IsNullOrEmpty(checkOutResponse.ErrorMessage))
375 {
376 return new ServiceResult<ProgramCheckinResponse>(null, checkOutResponse.ErrorMessage);
377 }
378 }
379
380 var program = Encoding.UTF8.GetString(checkOutResponse.Data.Maml.Select(i => (byte)i).ToArray());
381 var mamlDoc = XDocument.Parse(program);
382 foreach (var xProgram in mamlDoc.Descendants("Program"))
383 {
384 xProgram.Attribute("State").Value = "Stop";
385 foreach (var xCampaign in xProgram.Descendants("Campaign"))
386 {
387 xCampaign.Attribute("State").Value = "Stop";
388 foreach (var xCampaignElement in xCampaign.Descendants("CampaignElement"))
389 {
390 xCampaignElement.Attribute("State").Value = "Stop";
391 foreach (var xSchedule in xCampaignElement.Descendants("Schedule"))
392 {
393 xSchedule.Attribute("Status").Value = "Stopped";
394 }
395 }
396 }
397 }
398
399 var checkInResponse = await CheckInProgram(credentials, mamlDoc.ToString(), comment);
400 if (checkInResponse.ErrorMessage != "")
401 {
402 var undoProgramChangesResponse = await UndoProgramChanges(ticket, programId);
403 if (!string.IsNullOrWhiteSpace(undoProgramChangesResponse.ErrorMessage))
404 {
405 return new ServiceResult<ProgramCheckinResponse>(null, undoProgramChangesResponse.ErrorMessage);
406 }
407
408 return new ServiceResult<ProgramCheckinResponse>(null, "Something went wrong, please try again.");
409 }
410
411 return new ServiceResult<ProgramCheckinResponse>(checkInResponse.Data, "");
412 }
