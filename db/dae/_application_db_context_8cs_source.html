197 private readonly ISessionService _sessionService;
198 private ISession _session => _sessionService.Session;

200 public ApplicationDbContext(DbContextOptions<ApplicationDbContext> options, ISessionService sessionService)
201 : base(options)
202 {
203 _sessionService = sessionService;
204 }
205
206 private void OnBeforeSaveChanges()
207 {
208 AddTimestampsToEntities();
209 }
210
211 private void AddTimestampsToEntities()
212 {
213 var clientName = SessionExtensions.GetString(_session, "Client");
214 var clientId = SessionExtensions.GetInt32(_session, "ClientId");
215
216 foreach (var entity in from e in ChangeTracker.Entries<BaseEntity>()
217 where e.State == EntityState.Added
218 select e.Entity)
219 {
220
221 if (clientId != null)
222 {
223 entity.ClientId = clientId;
224 }
225 }
226
227 foreach (var entity in from e in ChangeTracker.Entries<BaseEntity>()
228 where e.State == EntityState.Modified
229 select e.Entity)
230 {
231 if (clientId != null)
232 {
233 entity.ClientId = clientId;
234 }
235 }
236
237 foreach (var entity in from e in ChangeTracker.Entries<IModified>()
238 where e.State == EntityState.Added
239 select e.Entity)
240 {
241 entity.DateCreated = DateTime.UtcNow;
242 entity.DeluxeLastModified = DateTime.UtcNow;
243 }
244
245 foreach (var entity in from e in ChangeTracker.Entries<IModified>()
246 where e.State == EntityState.Modified
247 select e.Entity)
248 {
249 entity.DeluxeLastModified = DateTime.UtcNow;
250 }
251 }
252 public override int SaveChanges()
253 {
254 OnBeforeSaveChanges();
255 return base.SaveChanges();
256 }
257
258 // For use in DatabaseSession, since otherwise AddTimestampsToEntities() would trigger an infinite loop because it relies on SessionService
259 public int SaveChangesWithoutTimestamps()
260 {
261 return base.SaveChanges();
262 }
263
264 public override Task<int> SaveChangesAsync(CancellationToken cancellationToken = default)
265 {
266 OnBeforeSaveChanges();
267 return base.SaveChangesAsync(cancellationToken);
268 }
269
270 public void DetachAllEntities()
271 {
272 var changedEntriesCopy = ChangeTracker.Entries()
273 .Where(
274 e => e.State is
275 EntityState.Added or
276 EntityState.Modified or
277 EntityState.Deleted
278 )
279 .ToList();
280
281 foreach (var entry in changedEntriesCopy)
282 {
283 entry.State = EntityState.Detached;
284 }
285 }
