44 public ServiceResult<CheckEmailDomainResponse> CheckEmailDomain(string email)
45 {
46 if (!Util.Email.IsValidEmail(email))
47 {
48 return new ServiceResult<CheckEmailDomainResponse>(null, "Email not valid format.");
49 }
50
51 var domain = email.Split("@").ElementAt(1);
52
53 if (dbContext.EmailDomains.Any(x => x.BlackListed && x.Domain == domain))
54 {
55 return new ServiceResult<CheckEmailDomainResponse>(new CheckEmailDomainResponse { Status = EmailDomainType.Blacklisted, Message = "This email domain is blacklisted." }, "");
56 }
57 else if (dbContext.EmailDomains.Any(x => x.Domain == domain))
58 {
59 return new ServiceResult<CheckEmailDomainResponse>(new CheckEmailDomainResponse { Status = EmailDomainType.Approved, Message = "Success" }, "");
60 }
61 else
62 {
63 return new ServiceResult<CheckEmailDomainResponse>(new CheckEmailDomainResponse { Status = EmailDomainType.EmailSPFApproved, Message = "Success but it needs to be SPF approved." }, "");
64 }
