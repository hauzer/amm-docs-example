162 private static ContactGroupVM RolloutGroups(this ContactGroupVM group, IEnumerable<ContactGroup> contactGroups, int recursionLevel = 0)
163 {
164 if (group is null || group.SegmentationFilter is null)
165 {
166 return group;
167 }
168
169 var filter = group.SegmentationFilter;
170 _ = group.RolloutBlank();
171
172 var criterias = filter.Criterias.Where(c => c.Field.StartsWith("group", StringComparison.CurrentCultureIgnoreCase));
173 foreach (var criteria in new List<SegmentationFilterCriteria>(criterias))
174 {
175 var groupsToRollOut = new List<ContactGroup>();
176 switch (criteria.Operation.ToLower())
177 {
178 case "in":
179 {
180 var groupToAdd = contactGroups.FirstOrDefault(g => g.Name.Equals(criteria.Value, StringComparison.CurrentCultureIgnoreCase));
181 if (groupToAdd is not null)
182 {
183 groupsToRollOut.Add(groupToAdd);
184 }
185 }
186
187 break;
188 case "contains":
189 {
190 var groupsToAdd = contactGroups.Where(g => g.Name.Contains(criteria.Value, StringComparison.CurrentCultureIgnoreCase) && g.Name != group.Name);
191 if (groupsToAdd is not null)
192 {
193 groupsToRollOut.AddRange(groupsToAdd);
194 }
195 }
196
197 break;
198 case "notcontains":
199 {
200 var groupsToAdd = contactGroups.Where(g => !g.Name.Contains(criteria.Value, StringComparison.CurrentCultureIgnoreCase) && g.Name != group.Name);
201 if (groupsToAdd is not null)
202 {
203 groupsToRollOut.AddRange(groupsToAdd);
204 }
205 }
206
207 break;
208 case "blank":
209 {
210
211 }
212
213 break;
214 default:
215 {
216 throw new ContactManagerException($"Invalid operation {criteria.Operation}");
217 }
218 }
219
220 if (groupsToRollOut.Count == 0 || recursionLevel > 100)
221 {
222 if (recursionLevel > 100)
223 {
224 Log.Warning("Segmentation filter too complex, ceasing further group rollout.");
225 }
226
227 var rolledOutGroupFilter = new SegmentationFilter
228 {
229 CriteriaJoinOperator = "1"
230 };
231 rolledOutGroupFilter.Criterias.Add(new SegmentationFilterCriteria("AccountID", "Equal", "", 1));
232
233 criteria.Field = "AccountID";
234 criteria.Operation = "Equal";
235 criteria.Value = "";
236 }
237 else
238 {
239 var rolledOutGroups = new List<ContactGroupVM>();
240 foreach (var groupToRollOut in groupsToRollOut)
241 {
242 var rolledOutGroupVm = new ContactGroupVM
243 {
244 Name = groupToRollOut.Name,
245 SegmentationFilter = groupToRollOut.Filter?.DeserializeXML<SegmentationFilter>() ?? new SegmentationFilter()
246 };
247 rolledOutGroups.Add(rolledOutGroupVm.RolloutGroups(contactGroups.Where(cg => !groupsToRollOut.Any(g => g.Name == cg.Name)), recursionLevel + 1));
248 }
249
250 var combinedFilter = rolledOutGroups.GetCombinedFilterAndFields().SummaryFilter;
251 filter.ReplaceRowWithFilter(combinedFilter, criteria.Row);
252 }
253 }
254
255 return group;

300 public static void Renumber(this SegmentationFilter filter, int start, int step)
301 {
302 if (filter is null || filter.CriteriaJoinOperator.IsNullOrEmpty())
303 {
304 return;
305 }
306
307 var param = new Dictionary<string, string>();
308 foreach (var criteria in filter.Criterias.OrderBy(r => r.Row))
309 {
310 var currRowNo = criteria.Row.ToString();
311 criteria.Row = start;
312 param.Add(currRowNo, start.ToString());
313 start += step;
314 }
315
316 filter.CriteriaJoinOperator = filter.CriteriaJoinOperator.ReplaceNumbers(param);

318 public static void Merge(this SegmentationFilter destination, SegmentationFilter source, string mergeOperator)
319 {
320 if (source is null)
321 {
322 return;
323 }
324
325 var lastRow = destination.Criterias.Max(r => r.Row);
326 source.Renumber(lastRow + 1, 1);
327 destination.Criterias.AddRange(source.Criterias);
328 if (!source.CriteriaJoinOperator.HasFullBrackets('(', ')'))
329 {
330 source.CriteriaJoinOperator = '(' + source.CriteriaJoinOperator + ')';
331 }
332
333 if (!destination.CriteriaJoinOperator.HasFullBrackets('(', ')'))
334 {
335 destination.CriteriaJoinOperator = '(' + source.CriteriaJoinOperator + ')';
336 }
337
338 destination.CriteriaJoinOperator += ' ' + mergeOperator + ' ' + source.CriteriaJoinOperator;

340 public static void ReplaceRowWithFilter(this SegmentationFilter destination, SegmentationFilter source, int row)
341 {
342 if (destination is null || destination.CriteriaJoinOperator.IsNullOrEmpty() || source is null || source.CriteriaJoinOperator.IsNullOrEmpty())
343 {
344 return;
345 }
346
347 var lastRow = destination.Criterias.Max(c => c.Row);
348 source.Renumber(lastRow + 1, 1);
349 var rowToReplace = row.ToString();
350 var rowToRemove = destination.Criterias.FirstOrDefault(r => r.Row.Equals(row)) ?? throw new ContactManagerException("Invalid SegmentationFilter.");
351
352 _ = destination.Criterias.Remove(rowToRemove);
353
354 // replace old row from destination.CriteriaJoinOperator with new CriteriaJoinOperator
355 destination.CriteriaJoinOperator = destination.CriteriaJoinOperator.ReplaceNumbers(new Dictionary<string, string> { { rowToReplace, source.CriteriaJoinOperator } });
356 destination.Criterias.AddRange(source.Criterias);

359 public static GetCombinedFilterAndFieldsResult GetCombinedFilterAndFields(this IEnumerable<ContactGroupVM> contactGroups)
360 {
361 var fieldNames = new List<string>();
362 SegmentationFilter summaryFilter = null;
363 if (contactGroups != null && contactGroups.Any())
364 {
365 var hasAllContactsFilter = false;
366 foreach (var contactGroup in contactGroups)
367 {
368 var filter = contactGroup.SegmentationFilter;
369 if (filter is null || filter.CriteriaJoinOperator.IsNullOrEmpty() || filter.Criterias is null)
370 {
371 continue;
372 }
373
374 // checking for special case when filter.CriteriaJoinOperator = "|" or = "&";
375 if (filter.CriteriaJoinOperator.Equals("|") || filter.CriteriaJoinOperator.Equals("&") || filter.CriteriaJoinOperator.Equals("&amp;"))
376 {
377 if (filter.Criterias == null || filter.Criterias.Count == 0)
378 {
379 hasAllContactsFilter = true;
380 break;
381 }
382
383 var operation = filter.CriteriaJoinOperator;
384 filter.CriteriaJoinOperator = "(";
385 foreach (var item in filter.Criterias)
386 {
387 filter.CriteriaJoinOperator += item.Row.ToString();
388 }
389
390 filter.CriteriaJoinOperator += ")";
391 }
392
393 var fields = filter.Criterias.Where(c => !(c.Field.Equals("accountid", StringComparison.CurrentCultureIgnoreCase) && c.Operation.Equals("equal", StringComparison.CurrentCultureIgnoreCase) && c.Value == ""));
394 fieldNames.AddRange(fields.Select(c => c.Field).ToList());
395 if (summaryFilter == null)
396 {
397 summaryFilter = filter;
398 }
399 else
400 {
401 summaryFilter.Merge(filter, "|");
402 }
403 }
404
405 fieldNames = fieldNames.Distinct().ToList();
406
407 if (hasAllContactsFilter)
408 {
409 summaryFilter = new SegmentationFilter
410 {
411 CriteriaJoinOperator = "|"
412 };
413 }
414 }
415
416 return new GetCombinedFilterAndFieldsResult()

418 FieldNames = fieldNames,
419 SummaryFilter = summaryFilter,

428 public static Func<ContactVM, bool> GetContactMatcher(this SegmentationFilter filter, string joinOperator = null, Func<ContactVM, bool> previousMatcher = null)
429 {
430 joinOperator ??= filter.CriteriaJoinOperator;
431
432 if (joinOperator.IsNullOrEmpty())
433 {
434 return null;
435 }
436
437 for (var i = 0; i < joinOperator.Length; ++i)
438 {
439 if (char.IsWhiteSpace(joinOperator[i]))
440 {
441 continue;
442 }
443 else if (joinOperator[i] == '(')
444 {
445 ++i;
446 var j = i;
447 var parenthesesBalance = 1;
448
449 while (i < joinOperator.Length && parenthesesBalance > 0)
450 {
451 if (joinOperator[i] == '(')
452 {
453 ++parenthesesBalance;
454 }
455 else if (joinOperator[i] == ')')
456 {
457 --parenthesesBalance;
458 }
459
460 ++i;
461 };
462
463 if (parenthesesBalance != 0)
464 {
465 throw new ContactManagerException($"Unbalanced parentheses in contact filter: {parenthesesBalance}");
466 }
467
468 var subMatcher = GetContactMatcher(filter, joinOperator.Substring(j, i - (j + 1)), previousMatcher);
469 return GetContactMatcher(filter, joinOperator[i..], subMatcher) ?? subMatcher;
470 }
471 else if (char.IsDigit(joinOperator[i]))
472 {
473 var numberSb = new StringBuilder();
474 do
475 {
476 _ = numberSb.Append(joinOperator[i]);
477 ++i;
478 }
479 while (i < joinOperator.Length && char.IsDigit(joinOperator[i]));
480 var number = numberSb.ToString();
481
482 Func<ContactVM, bool> matcher = null;
483 var criteria = filter.Criterias.First(c => c.Row == int.Parse(number));
484 string getContactFieldValue(ContactVM c) => c.GetPropertyValueIgnoreCase(criteria.Field)?.ToString().ToLower() ?? "";
485 switch (criteria.Operation.ToLower())
486 {
487 case "equal":
488 matcher = c => getContactFieldValue(c).Equals(criteria.Value, StringComparison.CurrentCultureIgnoreCase);
489 break;
490 case "notequal":
491 matcher = c => !getContactFieldValue(c).Equals(criteria.Value, StringComparison.CurrentCultureIgnoreCase);
492 break;
493 case "contains":
494 matcher = c => getContactFieldValue(c).Contains(criteria.Value, StringComparison.CurrentCultureIgnoreCase);
495 break;
496 case "notcontains":
497 matcher = c => !getContactFieldValue(c).Contains(criteria.Value, StringComparison.CurrentCultureIgnoreCase);
498 break;
499 case "startswith":
500 matcher = c => getContactFieldValue(c).StartsWith(criteria.Value, StringComparison.CurrentCultureIgnoreCase);
501 break;
502 case "endswith":
503 matcher = c => getContactFieldValue(c).EndsWith(criteria.Value.ToLower());
504 break;
505 case "from":
506 {
507 if (DateTime.TryParse(criteria.Value, out var criteriaValue))
508 {
509 matcher = c =>
510 {
511 if (DateTime.TryParse(getContactFieldValue(c), out var contactValue))
512 {
513 return contactValue > criteriaValue;
514 }
515
516 return false;
517 };
518 }
519 else
520 {
521 return _ => false;
522 }
523 }
524
525 break;
526 case "to":
527 {
528 if (DateTime.TryParse(criteria.Value, out var criteriaValue))
529 {
530 matcher = c =>
531 {
532 if (DateTime.TryParse(getContactFieldValue(c), out var contactValue))
533 {
534 return contactValue < criteriaValue;
535 }
536
537 return false;
538 };
539 }
540 else
541 {
542 return _ => false;
543 }
544 }
545
546 break;
547 default:
548 throw new ContactManagerException($"Invalid criteria operation {criteria.Operation}");
549 }
550
551 return GetContactMatcher(filter, joinOperator[i..], matcher) ?? matcher;
552 }
553 else if (joinOperator[i] is '|' or '&')
554 {
555 var logicalOperator = joinOperator[i];
556 ++i;
557
558 if (previousMatcher is null)
559 {
560 throw new ContactManagerException($"Logical operator '{logicalOperator}' doesn't have a left-hand expression");
561 }
562
563 var nextMatcher = GetContactMatcher(filter, joinOperator[i..], previousMatcher) ?? throw new ContactManagerException($"Logical operator '{logicalOperator}' doesn't have a right-hand expression");
564
565 if (logicalOperator == '|')
566 {
567 return c => previousMatcher(c) || nextMatcher(c);
568 }
569 else
570 {
571 return c => previousMatcher(c) && nextMatcher(c);
572 }
573 }
574
575 throw new ContactManagerException($"Unrecognized character in contact filter: '{joinOperator[i]}'");
576 }
577
578 throw new NotImplementedException();
579 }
