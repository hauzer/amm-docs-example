43 public async Task<ServiceResult<string>> DeactivateUser(DeactivateVM deactivateVM)
44 {
45 ApplicationUser user = null;
46
47 if (System.Guid.TryParse(deactivateVM.UserId, out var userId))
48 {
49 user = ApplicationDbContext.Users.FirstOrDefault(x => x.Id == deactivateVM.UserId);

56 if (user == null)

58 return new ServiceResult<string>(null, "User not found.");
59 }
60
61 if (deactivateVM.Deactivate)
62 {
63 if (long.TryParse(user.MFAccountID, out var accountId))
64 {
65 var authenticateResult = await _mindfireService.userService.Authenticate(accountId);
66 // Stop All campaigns
67 if (string.IsNullOrWhiteSpace(authenticateResult.ErrorMessage))
68 {
69 var credentials = _mindfireService.requestCredentialsTemplate.NewInstance(authenticateResult.Data?.Credentials?.PartnerTicket, authenticateResult.Data?.Credentials?.Ticket);
70 var getProgramListResult = await _mindfireService.programService.GetProgramList(credentials, 1);
71 if (!string.IsNullOrWhiteSpace(getProgramListResult.ErrorMessage))
72 {
73 return new ServiceResult<string>(null, getProgramListResult.ErrorMessage);
74 }
75
76 var programList = getProgramListResult.Data.ProgramList.Select(pl => pl.ProgramID).ToList();
77 var response = await _userCampaignService.StopCampaigns(programList, user);
78 }
79 else
80 {
81 return new ServiceResult<string>(null, "Unable to authenticate user");
82 }
83
84 var deactivateUserResult = await _mindfireService.accountService.AccountSetSuspended(Convert.ToInt64(user.MFAccountID), user.MFTicket);
85 if (!string.IsNullOrWhiteSpace(deactivateUserResult.ErrorMessage))
86 {
87 return new ServiceResult<string>(null, deactivateUserResult.ErrorMessage);
88 }
89
90 user.Enabled = false;
91 if (user.AccountStatus != null)
92 {
93 var inactiveStatus = await _accountStatusService.Single(x => x.Name == "Inactive");
94 user.AccountStatusId = inactiveStatus.Data.Id;
95 }
96 }
97 else
98 {
99 return new ServiceResult<string>(null, "Invalid User MFAccountID.");
100 }
101 }
102 else
103 {
104 var activateUserResult = await _mindfireService.accountService.AccountSetActive(Convert.ToInt64(user.MFAccountID), user.MFTicket);
105 if (!string.IsNullOrWhiteSpace(activateUserResult.ErrorMessage))
106 {
107 return new ServiceResult<string>(null, activateUserResult.ErrorMessage);
108 }
109
110 user.Enabled = true;
111 if (user.AccountStatus != null)
112 {
113 var activeStatus = await _accountStatusService.Single(x => x.Name == "Active");
114 user.AccountStatusId = activeStatus.Data.Id;
115 }
116 }
117
118 _ = ApplicationDbContext.SaveChanges();
119 return new ServiceResult<string>(null, "");
