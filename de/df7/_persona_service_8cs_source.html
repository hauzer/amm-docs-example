59 public async Task<ServiceResult<IPaginate<PMSPersonaDTO>>> GetPersonasForProfile(string userId, FilterConfiguration filterConfiguration = null)
60 {
61 if (string.IsNullOrEmpty(userId))
62 {
63 userId = _sessionService.SessionGetUserId();
64 }
65
66 var personasRes = await _personaMicroservice.GetPersonaInstancesForUser_Simple(userId, "", "", true, "");
67 if (!string.IsNullOrEmpty(personasRes.ErrorMessage))
68 {
69 return new ServiceResult<IPaginate<PMSPersonaDTO>>(null, personasRes.ErrorMessage);
70 }
71
72 var personaFilterd = personasRes.Data.ToList();
73 var personaVMs = new List<PMSPersonaDTO>();
74 //TODO: Istead retreiving one by one persona retreive full persona list
75 foreach (var item in personaFilterd)

84 public async Task<ServiceResult<PMSPersonaDTO>> GetDefaultPersonaForProfile(string userId, Audience audience)
85 {
86 if (string.IsNullOrEmpty(userId))
87 {
88 var user = await GetUser();
89 userId = user.Id;
90 }
91
92 var personasRes = await _personaMicroservice.GetPersonaInstancesForUser_Simple(userId, "", "", true, "");
93
94 if (!string.IsNullOrEmpty(personasRes.ErrorMessage))
95 {
96 return new ServiceResult<PMSPersonaDTO>(null, personasRes.ErrorMessage);
97 }
98
99 if (personasRes.Data.Count == 0)
100 {
101 return new ServiceResult<PMSPersonaDTO>(null, "User has no personas");
102 }
103
104 var persona = personasRes.Data.FirstOrDefault(x => x.IsDefault) ?? personasRes.Data.FirstOrDefault(x => x.isActive);
105 var fullPersona = await _personaMicroservice.GetPersonaInstance(persona.PersonaInstanceId.ToString(), userId);
106 var personaVM = fullPersona.Data;
107
108 //check if audience is individual or company
109 switch (audience)
110 {
111 case Audience.Individual:
112 if (personaVM.Data.GetValueOrDefault("PersonaType").Equals("company", System.StringComparison.CurrentCultureIgnoreCase))
113 {
114 var personaInstance = await FindPersonaByType(personasRes.Data, "Individual");
115 if (!string.IsNullOrEmpty(personaInstance.ErrorMessage))
116 {
117 return new ServiceResult<PMSPersonaDTO>(null, personaInstance.ErrorMessage);
118 }
119
120 personaVM = personaInstance.Data;
121 }
122
123 break;
124 case Audience.Company:
125 if (personaVM.Data.GetValueOrDefault("PersonaType").Equals("individual", System.StringComparison.CurrentCultureIgnoreCase))
126 {
127 var personaInstance = await FindPersonaByType(personasRes.Data, "Company");
128 if (!string.IsNullOrEmpty(personaInstance.ErrorMessage))
129 {
130 return new ServiceResult<PMSPersonaDTO>(null, personaInstance.ErrorMessage);
131 }
132
133 personaVM = personaInstance.Data;
134 }
135
136 break;
137 case Audience.Both:
138 default:
139 break;
140 }
141
142 return new ServiceResult<PMSPersonaDTO>(personaVM, "");
143 }
