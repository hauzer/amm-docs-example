62 public void SetFutureDateAndRecurrentMessage()
63 {
64 var csh = new CampaignScheduleHelper();
65 //Get next future date and recurrence message
66 var result = Recurrences?.Select(r => new { Dates = csh.GetScheduledDates(r, Start), RecurrenceId = r.Id })
67 .SelectMany(a => a.Dates.Select(date => new { Date = date, a.RecurrenceId }))
68 .Where(f => DateTime.UtcNow < f.Date).OrderBy(o => o.Date).FirstOrDefault();
69 if (result is null)
70 {
71 return;
72 }
73
74 ScheduleConfig ??= new ScheduleConfigVM();
75 ScheduleConfig.ScheduledLaunchDate = result.Date;
76 ScheduleConfig.RecurrenceMessage = CampaignScheduleHelper.GetRecurrenceMessage(Recurrences.FirstOrDefault(x => x.Id == result.RecurrenceId));
77 }
